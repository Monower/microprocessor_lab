MACRO:

.MODEL SMALL
.STACK 100H


ADDITION MACRO A,B
    MOV BL,A 
    ADD BL,B
    
    MOV RESULT,BL
ENDM

.DATA
    RESULT DB ?
    N1 DB ?
    N2 DB ?
    STR1 DB 0AH,0DH, 'ENTER THE FIRST NUMBER: $'
    STR2 DB 0AH,0DH, 'ENTER THE SECOND NUMBER: $'
    STR3 DB 0AH,0DH, 'THE SUM IS: $'
    

.CODE
MAIN PROC
    
    MOV AX,@DATA
    MOV DS,AX
    
    MOV AH,9
    MOV DX,OFFSET STR1
    INT 21H
    
    MOV AH,1
    INT 21H
    
    MOV N1,AL
    SUB N1,30H
    
    MOV AH,9
    MOV DX,OFFSET STR2
    INT 21H
    
    MOV AH,1
    INT 21H
    
    MOV N2,AL
    SUB N2,30H
    
    ADDITION N1,N2
    
    MOV AH,9
    MOV DX,OFFSET STR3
    INT 21H
    
    ADD RESULT,30H
    
    MOV DL,RESULT
    MOV AH,2
    INT 21H
    
    
    
    
    


MAIN ENDP


END MAIN
